```javascript
class TimeDisplay {
  constructor(selector) {
    this.element = document.querySelector(selector);
    this.init();
  }
  
  init() {
    this.update();
    setInterval(() => this.update(), 1000);
  }
  
  update() {
    const now = new Date();
    const timeString = now.toLocaleTimeString('zh-CN', {
      hour12: false,
      hour: '2-digit',
      minute: '2-digit',
      second: '2-digit'
    });
    
    if (this.element) {
      this.element.textContent = timeString;
    }
  }
}
```

```javascript
class ParticleSystem {
  constructor() {
    this.particles = [];
    this.maxParticles = 50;
    this.init();
  }
  
  init() {
    setInterval(() => this.createParticle(), 200);
    setInterval(() => this.cleanup(), 1000);
  }
  
  createParticle() {
    if (this.particles.length >= this.maxParticles) return;
    
    const particle = document.createElement('div');
    particle.className = 'particle particle-animate';
    particle.style.left = Math.random() * window.innerWidth + 'px';
    particle.style.animationDelay = Math.random() * 2 + 's';
    particle.style.animationDuration = (Math.random() * 4 + 4) + 's';
    
    document.body.appendChild(particle);
    this.particles.push(particle);
    
    setTimeout(() => {
      this.removeParticle(particle);
    }, 8000);
  }
  
  removeParticle(particle) {
    const index = this.particles.indexOf(particle);
    if (index > -1) {
      this.particles.splice(index, 1);
      particle.remove();
    }
  }
  
  cleanup() {
    this.particles = this.particles.filter(p => document.body.contains(p));
  }
}
```

```javascript
        e.currentTarget.classList.add('active');
        
        // æ·»åŠ ç‚¹å‡»æ³¢çº¹æ•ˆæœ

      });
    });
  }
  
  bindCardEvents() {
    document.querySelectorAll('.app-card, .featured-card').forEach(card => {
      card.addEventListener('mouseenter', () => {
        card.style.setProperty('--hover-scale', '1.05');
      });
      
      card.addEventListener('mouseleave', () => {
        card.style.removeProperty('--hover-scale');
      });
      
      card.addEventListener('click', (e) => {
        this.createClickEffect(e.currentTarget);
      });
    });
  }
  
  createRipple(e) {
    const ripple = document.createElement('span');
    const rect = e.currentTarget.getBoundingClientRect();
    const size = Math.max(rect.width, rect.height);
    const x = e.clientX - rect.left - size / 2;
    const y = e.clientY - rect.top - size / 2;
    
    ripple.style.cssText = `
      position: absolute;
      border-radius: 50%;
      background: rgba(255, 0, 102, 0.3);
      transform: scale(0);
      animation: ripple 0.6s linear;
      left: ${x}px;
      top: ${y}px;
      width: ${size}px;
      height: ${size}px;
      pointer-events: none;
    `;
    
    e.currentTarget.appendChild(ripple);
    setTimeout(() => ripple.remove(), 600);
  }
  
  createClickEffect(element) {
    element.style.transform = 'scale(0.95)';
    setTimeout(() => {
      element.style.transform = '';
    }, 150);
  }
}
```
